 % بسته و دستوری برای ریست کردن شماره پانویس‌ها در هر صفحه
\RequirePackage{zref-perpage}
\zmakeperpage{footnote}
% بسته‌ای برای تبدیل ارجاع‌های کتاب به لینک‌های کلیک‌شدنی

\RequirePackage{hyperref}
\hypersetup{
    colorlinks=true,
    citecolor=blue,
    filecolor=black,
    linkcolor=red,
    urlcolor=green,
}
% بسته‌ای برای گزینه‌های بیشتر در ارجاع‌نویسی
% گزینه‌ای برای بازگشت به محل ارجاع
% گزینه‌ای برای لیست مراجع و منابع در هر فصل
\RequirePackage[backend=biber,refsection=chapter,backref=true,style=numeric,sorting=none]{biblatex}
% تنظیم متن ارجاع بازگشتی
\DefineBibliographyStrings{english}{
    backrefpage={\rl{ارجاع‌شده در صفحهٔ}},
    backrefpages={\rl{ارجاع‌شده در صفحه‌های}}
}
